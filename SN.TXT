"""
ServiceNow → Jira Story Creator  (Team-Aware Version)
======================================================
Filters ServiceNow records to ONLY your team's tickets using:
  1. Assignment Group  (primary — recommended)
  2. Assigned-To list  (individual members)
  3. Category/Subcategory
  4. CI (Configuration Item) ownership
  5. Watched-by / Opened-by your team

Requirements:
    pip install requests python-dotenv

Environment Variables (.env):
    SNOW_INSTANCE         = your-instance.service-now.com
    SNOW_USER             = servicenow_api_user
    SNOW_PASS             = servicenow_api_password
    SNOW_TABLE            = incident
    SNOW_LIMIT            = 50

    # ── Team Identity ──────────────────────────────────────────
    TEAM_ASSIGNMENT_GROUP = Cloud Platform Engineering          ← group name in SNOW
    TEAM_MEMBERS          = alice@co.com,bob@co.com,sunita@co.com
    TEAM_CATEGORIES       = AI Platform,Data Engineering,Cloud
    TEAM_CIS              = SafeAlign,AI-Gateway,OpenShift-Cluster

    JIRA_BASE_URL         = https://your-org.atlassian.net
    JIRA_USER             = jira_email@company.com
    JIRA_API_TOKEN        = your_jira_api_token
    JIRA_PROJECT_KEY      = PROJ
"""

import os
import sys
import json
import logging
import requests
from datetime import datetime
from dotenv import load_dotenv

load_dotenv()

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s",
    handlers=[
        logging.StreamHandler(sys.stdout),
        logging.FileHandler(f"snow_jira_{datetime.now().strftime('%Y%m%d_%H%M%S')}.log"),
    ],
)
log = logging.getLogger(__name__)


# ── Team Config ────────────────────────────────────────────────────────────────
class TeamConfig:
    """
    Defines WHO your team is in ServiceNow terms.
    All values come from .env — no hardcoding needed.
    """

    # Primary: The assignment group name exactly as it appears in ServiceNow
    ASSIGNMENT_GROUP = os.getenv("TEAM_ASSIGNMENT_GROUP", "")

    # Secondary: Individual team member emails/usernames (comma-separated)
    MEMBERS = [m.strip() for m in os.getenv("TEAM_MEMBERS", "").split(",") if m.strip()]

    # Optional: Categories your team owns
    CATEGORIES = [c.strip() for c in os.getenv("TEAM_CATEGORIES", "").split(",") if c.strip()]

    # Optional: CI names (apps/services) your team owns
    CIS = [ci.strip() for ci in os.getenv("TEAM_CIS", "").split(",") if ci.strip()]

    # Filter strategy: "group" | "members" | "category" | "ci" | "any"
    # "any" = match if ANY of the above signals match (broadest)
    STRATEGY = os.getenv("TEAM_FILTER_STRATEGY", "group")


# ── ServiceNow Query Builder ───────────────────────────────────────────────────
class SnowQueryBuilder:
    """
    Builds a ServiceNow encoded query string that scopes records to your team.
    ServiceNow uses '^' for AND and '^OR' for OR between conditions.
    """

    BASE_FILTER = "active=true^stateNOT IN6,7"   # exclude Resolved + Closed

    @staticmethod
    def by_group(group_name: str) -> str:
        """
        BEST approach. Matches assignment_group.name = 'Your Team'.
        In SNOW encoded query: assignment_group.name=Cloud Platform Engineering
        """
        return f"{SnowQueryBuilder.BASE_FILTER}^assignment_group.name={group_name}"

    @staticmethod
    def by_members(emails: list[str]) -> str:
        """
        Matches assigned_to.email IN (alice, bob, sunita).
        Good fallback when no formal group exists.
        """
        if not emails:
            return SnowQueryBuilder.BASE_FILTER
        conditions = "^ORassigned_to.email=".join(emails)
        return f"{SnowQueryBuilder.BASE_FILTER}^assigned_to.email={conditions}"

    @staticmethod
    def by_category(categories: list[str]) -> str:
        """
        Matches category IN ('AI Platform', 'Cloud').
        Useful when your team owns a specific service category.
        """
        if not categories:
            return SnowQueryBuilder.BASE_FILTER
        cat_filter = "^ORcategory=".join(categories)
        return f"{SnowQueryBuilder.BASE_FILTER}^category={cat_filter}"

    @staticmethod
    def by_ci(ci_names: list[str]) -> str:
        """
        Matches cmdb_ci.name IN ('SafeAlign', 'AI-Gateway').
        Best for platform teams that own specific services/apps.
        """
        if not ci_names:
            return SnowQueryBuilder.BASE_FILTER
        ci_filter = "^ORcmdb_ci.name=".join(ci_names)
        return f"{SnowQueryBuilder.BASE_FILTER}^cmdb_ci.name={ci_filter}"

    @staticmethod
    def combined_any(group: str, members: list[str], categories: list[str], cis: list[str]) -> str:
        """
        OR across all signals — broadest net.
        Catches tickets assigned to group OR any member OR any owned CI.
        """
        parts = []
        if group:
            parts.append(f"assignment_group.name={group}")
        for m in members:
            parts.append(f"assigned_to.email={m}")
        for c in categories:
            parts.append(f"category={c}")
        for ci in cis:
            parts.append(f"cmdb_ci.name={ci}")

        if not parts:
            return SnowQueryBuilder.BASE_FILTER

        return SnowQueryBuilder.BASE_FILTER + "^NQ" + "^OR".join(parts)

    @classmethod
    def build(cls) -> str:
        """Select the right strategy based on TEAM_FILTER_STRATEGY env var."""
        strategy = TeamConfig.STRATEGY.lower()

        if strategy == "group":
            if not TeamConfig.ASSIGNMENT_GROUP:
                raise ValueError("TEAM_ASSIGNMENT_GROUP is required for strategy=group")
            q = cls.by_group(TeamConfig.ASSIGNMENT_GROUP)

        elif strategy == "members":
            if not TeamConfig.MEMBERS:
                raise ValueError("TEAM_MEMBERS is required for strategy=members")
            q = cls.by_members(TeamConfig.MEMBERS)

        elif strategy == "category":
            if not TeamConfig.CATEGORIES:
                raise ValueError("TEAM_CATEGORIES is required for strategy=category")
            q = cls.by_category(TeamConfig.CATEGORIES)

        elif strategy == "ci":
            if not TeamConfig.CIS:
                raise ValueError("TEAM_CIS is required for strategy=ci")
            q = cls.by_ci(TeamConfig.CIS)

        elif strategy == "any":
            q = cls.combined_any(
                TeamConfig.ASSIGNMENT_GROUP,
                TeamConfig.MEMBERS,
                TeamConfig.CATEGORIES,
                TeamConfig.CIS,
            )
        else:
            raise ValueError(f"Unknown TEAM_FILTER_STRATEGY: {strategy}")

        log.info(f"[SNOW QUERY] Strategy='{strategy}'  →  {q}")
        return q


# ── Config ─────────────────────────────────────────────────────────────────────
class Config:
    SNOW_BASE_URL  = f"https://{os.getenv('SNOW_INSTANCE')}/api/now"
    SNOW_USER      = os.getenv("SNOW_USER")
    SNOW_PASS      = os.getenv("SNOW_PASS")
    SNOW_TABLE     = os.getenv("SNOW_TABLE", "incident")
    SNOW_LIMIT     = int(os.getenv("SNOW_LIMIT", "50"))
    SNOW_FIELDS    = (
        "sys_id,number,short_description,description,"
        "priority,urgency,impact,state,category,"
        "assigned_to,assignment_group,caller_id,"
        "opened_at,sys_updated_on,cmdb_ci"
    )
    JIRA_BASE_URL  = os.getenv("JIRA_BASE_URL")
    JIRA_USER      = os.getenv("JIRA_USER")
    JIRA_API_TOKEN = os.getenv("JIRA_API_TOKEN")
    JIRA_PROJECT   = os.getenv("JIRA_PROJECT_KEY", "PROJ")

    @classmethod
    def validate(cls):
        required = {
            "SNOW_INSTANCE":  os.getenv("SNOW_INSTANCE"),
            "SNOW_USER":      cls.SNOW_USER,
            "SNOW_PASS":      cls.SNOW_PASS,
            "JIRA_BASE_URL":  cls.JIRA_BASE_URL,
            "JIRA_USER":      cls.JIRA_USER,
            "JIRA_API_TOKEN": cls.JIRA_API_TOKEN,
        }
        missing = [k for k, v in required.items() if not v]
        if missing:
            raise EnvironmentError(f"Missing env vars: {', '.join(missing)}")


# ── ServiceNow Client ──────────────────────────────────────────────────────────
class ServiceNowClient:
    def __init__(self):
        self.session = requests.Session()
        self.session.auth    = (Config.SNOW_USER, Config.SNOW_PASS)
        self.session.headers = {"Accept": "application/json", "Content-Type": "application/json"}

    def get_records(self, query: str) -> list[dict]:
        url    = f"{Config.SNOW_BASE_URL}/table/{Config.SNOW_TABLE}"
        params = {
            "sysparm_query":                  query,
            "sysparm_fields":                 Config.SNOW_FIELDS,
            "sysparm_limit":                  Config.SNOW_LIMIT,
            "sysparm_display_value":          "true",
            "sysparm_exclude_reference_link": "true",
        }
        log.info(f"Fetching records from ServiceNow table='{Config.SNOW_TABLE}' ...")
        resp = self.session.get(url, params=params, timeout=30)
        resp.raise_for_status()
        records = resp.json().get("result", [])
        log.info(f"  → {len(records)} team records retrieved.")
        return records

    def get_my_group_sys_id(self, group_name: str) -> str | None:
        """
        Optional helper: resolve group name → sys_id for stricter query.
        Useful if your group name has special characters.
        """
        url    = f"{Config.SNOW_BASE_URL}/table/sys_user_group"
        params = {"sysparm_query": f"name={group_name}", "sysparm_fields": "sys_id,name", "sysparm_limit": 1}
        resp = self.session.get(url, params=params, timeout=15)
        if resp.status_code == 200:
            results = resp.json().get("result", [])
            if results:
                return results[0]["sys_id"]
        return None


# ── Field Mapper ───────────────────────────────────────────────────────────────
PRIORITY_MAP = {
    "1 - Critical": {"name": "Highest"},
    "2 - High":     {"name": "High"},
    "3 - Moderate": {"name": "Medium"},
    "4 - Low":      {"name": "Low"},
    "5 - Planning": {"name": "Lowest"},
}

def map_to_jira(record: dict) -> dict:
    number      = record.get("number", "UNKNOWN")
    short_desc  = record.get("short_description", "(No summary)")
    description = record.get("description", "")
    priority    = PRIORITY_MAP.get(record.get("priority", ""), {"name": "Medium"})
    group       = record.get("assignment_group", "")
    assigned_to = record.get("assigned_to", "")
    category    = record.get("category", "")
    caller      = record.get("caller_id", "")
    opened_at   = record.get("opened_at", "")
    ci          = record.get("cmdb_ci", "")
    state       = record.get("state", "")

    jira_description = f"""
*ServiceNow Reference:* {number}
*Assignment Group:* {group}
*Assigned To (SNOW):* {assigned_to}
*Caller / Requester:* {caller}
*Category:* {category}
*Configuration Item (CI):* {ci}
*Priority (SNOW):* {record.get('priority', '')}
*State:* {state}
*Opened At:* {opened_at}

----
h3. Description

{description or '_No description provided._'}

----
_Auto-created from ServiceNow {number} | Group: {group}_
""".strip()

    return {
        "fields": {
            "project":     {"key": Config.JIRA_PROJECT},
            "issuetype":   {"name": "Story"},
            "summary":     f"[SNOW-{number}] {short_desc}",
            "description": jira_description,
            "priority":    priority,
            "labels":      [
                "servicenow-import",
                f"snow-{state.lower().replace(' ', '-')}",
                f"group-{group.lower().replace(' ', '-')[:30]}",
            ],
        }
    }


# ── Jira Client ────────────────────────────────────────────────────────────────
class JiraClient:
    def __init__(self):
        self.base = Config.JIRA_BASE_URL.rstrip("/")
        self.session = requests.Session()
        self.session.auth    = (Config.JIRA_USER, Config.JIRA_API_TOKEN)
        self.session.headers = {"Accept": "application/json", "Content-Type": "application/json"}

    def create_issue(self, payload: dict) -> dict:
        resp = self.session.post(f"{self.base}/rest/api/2/issue", json=payload, timeout=30)
        resp.raise_for_status()
        return resp.json()

    def issue_exists(self, snow_number: str) -> bool:
        jql  = f'project = "{Config.JIRA_PROJECT}" AND summary ~ "SNOW-{snow_number}"'
        resp = self.session.get(f"{self.base}/rest/api/2/search",
                                params={"jql": jql, "maxResults": 1}, timeout=15)
        return resp.status_code == 200 and resp.json().get("total", 0) > 0


# ── Orchestrator ───────────────────────────────────────────────────────────────
def run():
    Config.validate()

    snow  = ServiceNowClient()
    jira  = JiraClient()
    query = SnowQueryBuilder.build()

    records = snow.get_records(query)
    if not records:
        log.warning("No team records found. Check your TEAM_* env vars and SNOW query.")
        return

    results = {"created": [], "skipped": [], "failed": []}

    for rec in records:
        number = rec.get("number", "UNKNOWN")
        group  = rec.get("assignment_group", "?")
        try:
            if jira.issue_exists(number):
                log.info(f"  [SKIP]   {number} (group={group}) — already in Jira")
                results["skipped"].append(number)
                continue

            payload  = map_to_jira(rec)
            response = jira.create_issue(payload)
            jira_key = response.get("key", "?")
            log.info(f"  [CREATE] {number} → {jira_key}  (group={group})")
            results["created"].append({"snow": number, "jira": jira_key,
                                       "url": f"{Config.JIRA_BASE_URL}/browse/{jira_key}"})

        except requests.HTTPError as e:
            log.error(f"  [FAIL]   {number} — {e.response.status_code}: {e.response.text[:200]}")
            results["failed"].append({"snow": number, "error": str(e)})
        except Exception as e:
            log.error(f"  [FAIL]   {number} — {e}")
            results["failed"].append({"snow": number, "error": str(e)})

    log.info("\n" + "=" * 60)
    log.info(f"DONE  Created={len(results['created'])}  "
             f"Skipped={len(results['skipped'])}  Failed={len(results['failed'])}")

    out = f"results_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
    with open(out, "w") as f:
        json.dump(results, f, indent=2)
    log.info(f"Results → {out}")


if __name__ == "__main__":
    run()
