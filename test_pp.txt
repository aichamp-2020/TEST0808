def _font(px,bold=False):
    k=(px,bold)
    if k in _FC: return _FC[k]
    try: f=ImageFont.truetype(_FB if bold else _FR,max(6,px))
    except: f=ImageFont.truetype(_FF,max(6,px))
    _FC[k]=f; return f

# ── Spatial word (from PDF char stream) ─────────────────────────────
@dataclasses.dataclass
class SW:
    text:str; x0:float; top:float; x1:float; bottom:float
    size:float; bold:bool; color:Tuple

def _build_sw(page)->List[SW]:
    raw=page.chars
    if not raw: return []
    groups=[]; cur=[]
    for c in raw:
        if not c["text"] or not c["text"].strip():
            if cur: groups.append(cur); cur=[]; continue
        if not cur: cur.append(c); continue
        gap=c["x0"]-cur[-1]["x1"]; dy=abs(c["top"]-cur[-1]["top"])
        if dy<3 and gap<3: cur.append(c)
        else:
            if cur: groups.append(cur)
            cur=[c]
    if cur: groups.append(cur)
    words=[]
    for g in groups:
        text="".join(c["text"] for c in g)
        fn=g[0].get("fontname",""); bold="Bold" in fn or "bold" in fn
        col=g[0].get("non_stroking_color",(0.,0.,0.))
        if not isinstance(col,(list,tuple)): col=(0.,0.,0.)
        col=tuple(float(v) for v in list(col)[:3])
        if len(col)<3: col=col+(0.,)*(3-len(col))
        words.append(SW(text=text,
            x0=min(c["x0"] for c in g),top=min(c["top"] for c in g),
            x1=max(c["x1"] for c in g),bottom=max(c["bottom"] for c in g),
            size=float(g[0].get("size",9.0)),bold=bold,color=col))
    return words

def _find_sw(value:str,words:List[SW])->List[SW]:
    if not value.strip(): return []
    def _n(s): return re.sub(r"[\s\-./,()+@]","",s).lower()
    vn=_n(value)
    if not vn: return []
    for w in words:
        if w.text==value.strip() or _n(w.text)==vn: return [w]
    for win in range(1,min(9,len(words)+1)):
        for i in range(len(words)-win+1):
            grp=words[i:i+win]; tops=[w.top for w in grp]
            if max(tops)-min(tops)>22: continue
            combined=" ".join(w.text for w in grp); cn=_n(combined)
            if not cn or len(cn)<2: continue
            if vn in cn or (cn in vn and len(cn)>=max(4,len(vn)//2)) or value.strip() in combined:
                return grp
    vtoks=[t for t in re.split(r"[\s]+",value.strip()) if len(t)>2]
    if len(vtoks)>=2:
        found=[]; pool={i:w for i,w in enumerate(words)}
        for vt in vtoks:
            vtn=_n(vt)
            for i,w in list(pool.items()):
                if _n(w.text)==vtn or vtn in _n(w.text): found.append(w); del pool[i]; break
        if len(found)>=max(1,len(vtoks)//2):
            tops=[b.top for b in found]
            if not found or max(tops)-min(tops)<30: return found
    vc=value.strip().replace(" ","")
    for w in words:
        wc=w.text.replace(" ","")
        if len(wc)>=4 and (vc in wc or (wc in vc and len(wc)>=max(4,len(vc)//3))): return [w]
    return []

# ── Redact ───────────────────────────────────────────────────────────
def _c255(c): return tuple(min(255,max(0,int(v*255))) for v in c[:3])

def redact(img,boxes,masked,scale):
    if not boxes: return img
    draw=ImageDraw.Draw(img); PAD=1
    x0=min(b.x0 for b in boxes); y0=min(b.top for b in boxes)
    x1=max(b.x1 for b in boxes); y1=max(b.bottom for b in boxes)
    sz=boxes[0].size; bold=boxes[0].bold; col=boxes[0].color
    fpx=max(6,int(sz*scale)); tc=_c255(col)
    px0=int(x0*scale)-PAD; py0=int(y0*scale)-PAD
    px1=int(x1*scale)+PAD; py1=int(y1*scale)+PAD
    bw=px1-px0; bh=py1-py0
    draw.rectangle([px0,py0,px1,py1],fill=(255,255,255))
    if not masked.strip(): return img
    for a in range(fpx,5,-1):
        f=_font(a,bold)
        try: tw=draw.textbbox((0,0),masked.strip(),font=f)[2]
        except: tw=len(masked)*a*0.55
        if tw<=bw+6: fpx=a; break
    f=_font(fpx,bold)
    try: th=draw.textbbox((0,0),masked.strip(),font=f)[3]
    except: th=fpx
    ty=py0+max(0,(bh-th)//2)
    draw.text((px0+1,ty),masked.strip(),fill=tc,font=f)
    return img

    seen={}; res=[]
    def add(s,e,v):
        v=v.strip()
        if not v: return
        ws=v.split()
        while ws and ws[0].lower().rstrip(".:") in _OSTRIP: s+=len(ws[0])+1; ws=ws[1:]
        if not ws: return
        v=" ".join(ws); lv=v.lower()
        for sp in _OSTOP:
            if lv.startswith(sp) or lv==sp: return
        k=(s,s+len(v))

    for m in _KNOWN.finditer(text): add(m.start(),m.end(),m.group())
    for m in _OGEN.finditer(text): add(m.start(),m.end(),m.group())
    return res

@dataclasses.dataclass
class PM:
    label:str; original:str; masked:str; start:int; end:int

def _ov(s,e,occ): return any(s<oe and e>os for os,oe in occ)

def detect_pii(text:str,faker:Faker)->List[PM]:
    occ=[]; acc=[]
    def add(s,e,o,l):
        orig=o.strip()
        if not orig or len(orig)<2: return
        ld=len(o)-len(o.lstrip()); ss,ee=s+ld,s+ld+len(orig)
        if not _ov(ss,ee,occ): acc.append((ss,ee,orig,l)); occ.append((ss,ee))
    for l,rx in _CRX:
        for m in sorted(rx.finditer(text),key=lambda x:(x.start(),-(x.end()-x.start()))): add(m.start(),m.end(),m.group(),l)
    for s,e,v in sorted(_orgs(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"ORG")
    for s,e,v in sorted(_addrs(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"ADDRESS")
    for s,e,v in sorted(_label_names(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"NAME")
    for s,e,v in sorted(_persons(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"NAME")
    acc.sort(key=lambda x:x[0])
    return [PM(l,o,faker.mask(o,l),s,e) for s,e,o,l in acc]

# ── OCR ──────────────────────────────────────────────────────────────
def _ocr(img,faker):
    if not TESS: log.warning("pytesseract missing"); return img
    import pytesseract
    draw=ImageDraw.Draw(img)
    data=pytesseract.image_to_data(img,output_type=pytesseract.Output.DICT,lang="eng",config="--psm 6")
    n=len(data["text"])
    words=[(data["text"][i],data["left"][i],data["top"][i],data["width"][i],data["height"][i])
           for i in range(n) if data["text"][i].strip() and int(data["conf"][i])>30]
    if not words: return img
    full=" ".join(w[0] for w in words)
    matches=detect_pii(full,faker)
    cursor=0; wpos=[]
    for txt,l,t,w,h in words:
        idx=full.find(txt,cursor)
        if idx>=0: wpos.append((idx,idx+len(txt),l,t,w,h)); cursor=idx+len(txt)
    for m in matches:
        hits=[wp for wp in wpos if wp[0]<m.end and wp[1]>m.start]
        if not hits: continue
        x0=min(wp[2] for wp in hits); y0=min(wp[3] for wp in hits)
        x1=max(wp[2]+wp[4] for wp in hits); y1=max(wp[3]+wp[5] for wp in hits)
        bw=x1-x0; bh=y1-y0
        draw.rectangle([x0-1,y0-1,x1+1,y1+1],fill=(255,255,255))
        sz=max(6,int(bh*0.72))
        for a in range(sz,5,-1):
            f=_font(a,False)
            try: tw=draw.textbbox((0,0),m.masked,font=f)[2]
            except: tw=len(m.masked)*a*0.55
            if tw<=bw+4: sz=a; break
        draw.text((x0+1,y0+1),m.masked,fill=(0,0,0),font=_font(sz,False))
        log.debug("  OCR [%s] %r→%r",m.label,m.original,m.masked)
    return img

# ── Pipeline ─────────────────────────────────────────────────────────
def process_pdf(inp,out,scale=2.0,seed=42,report=True):
    inp=Path(inp); out=Path(out); faker=Faker(seed)
    log.info("Opening: %s  scale=%.1f",inp,scale)
    plb=pdfplumber.open(str(inp)); pdc=pdfium.PdfDocument(str(inp))
    pages=[]; summary={}; td=tdr=0
    for idx,pl in enumerate(plb.pages):
        log.info("  Page %d/%d",idx+1,len(plb.pages))
        img=pdc[idx].render(scale=scale,rotation=0).to_pil().convert("RGB")
        if len(pl.chars)<20:
            log.info("    Image-based → OCR")
            img=_ocr(img,faker); pages.append(img); continue
        raw=pl.extract_text(layout=False) or ""
        text=re.sub(r'([A-Za-z0-9._%+\-]+@[A-Za-z0-9.\-]+)\.\n([A-Za-z]{2,})',
                    lambda m:m.group(1)+"."+m.group(2),raw)
        matches=detect_pii(text,faker); summary[idx+1]=matches; td+=len(matches)
        if not matches: pages.append(img); continue
        sw=_build_sw(pl); drawn=0
        for m in matches:
            boxes=_find_sw(m.original,sw)
            if boxes:
                img=redact(img,boxes,m.masked,scale); drawn+=1
                log.debug("  [%-14s] %r→%r  sz=%.1f",m.label,m.original,m.masked,boxes[0].size)
            else:
                log.warning("  NO BOX [%s]: %r",m.label,m.original)
        tdr+=drawn; log.info("    detected=%d drawn=%d/%d",len(matches),drawn,len(matches))
        pages.append(img)
    plb.close(); pdc.close()
    log.info("Saving → %s",out)
    rgb=[i.convert("RGB") for i in pages]
    rgb[0].save(str(out),save_all=True,append_images=rgb[1:],resolution=150)
    log.info("DONE  det=%d drawn=%d pages=%d",td,tdr,len(pages))
    if report: _rpt(summary,tdr,td)
    return summary

def _rpt(s,d,t):
    print("\n"+"═"*70)
    print("  PII MASKING REPORT v6  ·  Spatial + OCR + Font Match")
    print("═"*70)
    c={}; g=0
    for pg,ms in sorted(s.items()):
        if ms:
            print(f"\n  Page {pg}:")
            for m in ms:
                lo=len(m.original.strip()); lm=len(m.masked.strip())
                print(f"    [{m.label:<16}] {m.original!r:<36} → {m.masked!r}  [{'✓' if lo==lm else f'✗{lo}>{lm}'}]")
                c[m.label]=c.get(m.label,0)+1; g+=1
    print("\n"+"─"*70); print(f"  Total: {g}   Drawn: {d}")
    for l,n in sorted(c.items(),key=lambda x:-x[1]):
        print(f"    {l:<22} {n:>3}  {'█'*min(n,35)}")
    print("═"*70)

def main():
    ap=argparse.ArgumentParser()
    ap.add_argument("input"); ap.add_argument("output")
    ap.add_argument("--scale",type=float,default=2.0)
    ap.add_argument("--seed",type=int,default=42)
    ap.add_argument("--no-report",action="store_true")
    ap.add_argument("--debug",action="store_true")
    args=ap.parse_args()

    if not Path(args.input).exists(): log.error("Not found: %s",args.input); sys.exit(1)

if __name__=="__main__":
    main()
