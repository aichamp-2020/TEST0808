class RealisticFaker:
    def __init__(self, seed=42):
        self._seed = seed
        self._cache: Dict[str,str] = {}

    def _rng(self, key:str) -> random.Random:
        h = int(hashlib.md5((str(self._seed)+key).encode()).hexdigest(),16)
        return random.Random(h & 0xFFFFFFFF)

    def _fit(self, word:str, n:int) -> str:
        w = word.strip()
        if not w: w = "X"
        if len(w) >= n: return w[:n].capitalize()
        pad = "abcdefghijklmnopqrstuvwxyz"
        while len(w) < n: w += pad[len(w) % len(pad)]
        return w.capitalize()

    def _fitlc(self, s:str, n:int) -> str:
        s = s.strip()
        if not s: s = "x"
        if len(s) >= n: return s[:n]
        while len(s) < n: s += s[-1]
        return s

    def fake_name(self, orig:str) -> str:
        rng = self._rng(orig)
        parts = orig.split()
        pool = _FIRST + _LAST
        return " ".join(self._fit(rng.choice(pool), len(p)) for p in parts)

    def fake_org(self, orig:str) -> str:
        """Generate realistic org name, word-for-word same length."""
        rng = self._rng(orig)
        parts = orig.split()
        pool = _ORG_PFX + _ORG_MID + _ORG_TYP + _CITIES
        result = []
        for p in parts:
            # Preserve suffixes like LLC, Inc, Ltd
            if p.upper() in ("LLC","INC","LTD","CORP","LP","LLP","PC","PLLC"):
                result.append(p)
            else:
                result.append(self._fit(rng.choice(pool), len(p)))
        return " ".join(result)

    def fake_ssn(self, orig:str) -> str:
        rng = self._rng(orig)
        sep = "." if "." in orig else (" " if " " in orig and "-" not in orig else "-")
        d = f"9{rng.randint(10,99):02d}{sep}{rng.randint(10,99):02d}{sep}{rng.randint(1000,9999):04d}"
        return _samelen(d, orig)

    def fake_email(self, orig:str) -> str:
        rng = self._rng(orig)
        # Strip any trailing newline/word noise
        clean = re.split(r'\n|\s{2,}', orig)[0].strip()
        m = re.match(r"(.+)@(.+)\.([A-Za-z]{2,})", clean)
        if not m: return self._scramble(orig)
        local = self._fitlc(
            f"{rng.choice(_FIRST).lower()}.{rng.choice(_LAST).lower()}", len(m.group(1)))
        dom   = self._fitlc(rng.choice(_EDOMAINS), len(m.group(2)))
        result = f"{local}@{dom}.{m.group(3)}"
        return _samelen(result, orig)

    def fake_phone(self, orig:str) -> str:
        rng = self._rng(orig)
        new = list(f"{rng.randint(201,989):03d}{rng.randint(200,999):03d}{rng.randint(1000,9999):04d}")
        res = list(orig); di = 0
        for i,c in enumerate(res):
            if c.isdigit() and di < len(new): res[i]=new[di]; di+=1
        return "".join(res)

    def fake_address(self, orig:str) -> str:
        rng = self._rng(orig)
        # Handle city/state/ZIP form
        city_state = re.match(r"([A-Za-z ]+),\s*([A-Z]{2})\s+(\d{5}(?:-\d{4})?)", orig)
        if city_state:
            fake_city  = self._fit(rng.choice(_CITIES), len(city_state.group(1).strip()))
            fake_state = rng.choice(_STATES)
            fake_zip   = self.fake_zip(city_state.group(3))
            return _samelen(f"{fake_city}, {fake_state} {fake_zip}", orig)
        # Street address
        toks = orig.split()
        out = []
        for i,t in enumerate(toks):
            if i==0 and re.match(r'^\d+$',t):
                n = rng.randint(10**(max(1,len(t)-1)), min(10**len(t)-1,99999))
                out.append(_samelen(str(n),t))
            else:
                out.append(self._fit(rng.choice(_STREETS+_STYPES+_CITIES), len(t)))
        return " ".join(out)

    def fake_account(self, orig:str) -> str:
        rng = self._rng(orig)
        res = list(orig)
        for i,c in enumerate(res):
            if c.isdigit(): res[i]=str(rng.randint(0,9))
        return "".join(res)

    def fake_date(self, orig:str) -> str:
        rng = self._rng(orig)
        y,m,d = rng.randint(1955,2005), rng.randint(1,12), rng.randint(1,28)
        if re.match(r"\d{4}[-/]\d{2}[-/]\d{2}", orig):
            r = f"{y:04d}-{m:02d}-{d:02d}"
        elif re.match(r"\d{2}[/\-]\d{2}[/\-]\d{4}", orig):
            r = f"{m:02d}/{d:02d}/{y:04d}"
        else:
            r = f"{y}-{m:02d}-{d:02d}"
        return _samelen(r, orig)

    def fake_zip(self, orig:str) -> str:
        rng = self._rng(orig)
        res = list(orig)
        for i,c in enumerate(res):
            if c.isdigit(): res[i]=str(rng.randint(0,9))
        return "".join(res)

    def fake_routing(self, orig:str) -> str:
        rng = self._rng(orig)
        new = [str(rng.randint(0,3))]+[str(rng.randint(0,9)) for _ in range(8)]
        res = list(orig); di=0
        for i,c in enumerate(res):
            if c.isdigit() and di<9: res[i]=new[di]; di+=1
        return "".join(res)

    def _scramble(self, orig:str) -> str:
        rng = self._rng(orig); res=[]
        for c in orig:
            if c.isupper():   res.append(rng.choice("BCDFGHJKLMNPQRSTVWXYZ"))
            elif c.islower(): res.append(rng.choice("bcdfghjklmnpqrstvwxyz"))
            elif c.isdigit(): res.append(str(rng.randint(2,9)))
            else: res.append(c)
        return "".join(res)

    def mask(self, orig:str, label:str) -> str:
        key = f"{label}::{orig}"
        if key in self._cache: return self._cache[key]
        fn = {
            "SSN":            self.fake_ssn,
            "EMAIL":          self.fake_email,
            "PHONE":          self.fake_phone,
            "CREDIT_CARD":    self.fake_account,
            "IP_ADDRESS":     lambda o: ".".join(str(self._rng(o).randint(1,253))
                                                 for _ in o.split(".")),
            "DATE_OF_BIRTH":  self.fake_date,
            "ROUTING_NUMBER": self.fake_routing,
            "ZIP_CODE":       self.fake_zip,
            "ADDRESS":        self.fake_address,
            "ACCOUNT_NUMBER": self.fake_account,
            "CLAIM_ID":       self.fake_account,
            "EIN":            self.fake_account,
            "NPI":            self.fake_account,
            "NAME":           self.fake_name,
            "ORG":            self.fake_org,
        }.get(label, self._scramble)
        r = _samelen(fn(orig), orig)
        self._cache[key] = r
        return r


def _samelen(r:str, orig:str) -> str:
    if not r: return orig
    if len(r) == len(orig): return r
    if len(r) > len(orig):  return r[:len(orig)]
    return r + r[-1] * (len(orig) - len(r))

_REGEX_PATTERNS = [
    # SSN
    ("SSN",            r"\b(?!000|666)\d{3}[-\s.]\d{2}[-\s.]\d{4}\b"),
    # Email — tight, no trailing context
    ("EMAIL",          r"\b[A-Za-z0-9._%+\-]+@[A-Za-z0-9.\-]+\.[A-Za-z]{2,6}\b"),
    # Phone US
    ("PHONE",          r"(?:\+1[-.\s]?)?(?:\(\d{3}\)|\d{3})[-.\s]?\d{3}[-.\s]?\d{4}\b"),
    # Credit card
    ("CREDIT_CARD",    r"\b(?:\d[ \-]?){13,16}\b"),
    # ISO date (YYYY-MM-DD) and slash date
    ("DATE_OF_BIRTH",  r"\b(?:(?:19|20)\d{2}[-/]\d{2}[-/]\d{2}|"
                       r"(?:0?[1-9]|1[0-2])[/\-]\d{2}[/\-](?:19|20)\d{2})\b"),
    # IP address
    ("IP_ADDRESS",     r"\b(?:(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\.){3}"
                       r"(?:25[0-5]|2[0-4]\d|[01]?\d\d?)\b"),
    # Account / member / policy / group / invoice IDs
    ("ACCOUNT_NUMBER", r"\b(?:ACCT|ACC|GRP|MBR|PLN|AUTH|CCN|NRB)[-\s]?[\w\-]{4,20}\b"),
    # Claim / RX / dental / vision IDs
    ("CLAIM_ID",       r"\b(?:MC|RX|DN|VS|PRM|CRD)-\d{4,8}\b"),
    # EIN  XX-XXXXXXX
    ("EIN",            r"\b\d{2}-\d{7}\b"),
    # ABA routing
    ("ROUTING_NUMBER", r"\b[0123]\d{8}\b"),
    # ZIP  NNNNN or NNNNN-NNNN
    ("ZIP_CODE",       r"\b\d{5}(?:-\d{4})?\b"),
    # NPI  10 digits
    ("NPI",            r"\b\d{10}\b"),
]
_COMPILED = [(lbl, re.compile(pat, re.I)) for lbl, pat in _REGEX_PATTERNS]

#  Detection Layer 2 — ORG (Organization / Company Name)
# ══════════════════════════════════════════════════════════════════════
# Anchor: any sequence of 1–5 TitleCase words ending in an org-type word
_ORG_SUFFIXES = (
    r"LLC|LLP|Inc\.?|Ltd\.?|Corp\.?|Co\.|LP\b|PC\b|PLLC"
    r"|Administration|Associates|Analytics|Solutions|Services|Group|Systems"
    r"|Consulting|Technologies|Enterprises|Holdings|Partners|Networks"
    r"|Healthcare|Diagnostics|Imaging|Practice|Dental|Vision|Pharmacy"
    r"|Rehab|Clinic|Center|Hospital|Medical|Laboratory|Laboratories"
    r"|Federal|Treasury|Financial|Bank|Insurance|Benefits"
)
_ORG_RE = re.compile(
    r"\b(?:[A-Z][A-Za-z&\./]{0,25}[ \t]){0,4}"  # 0-4 leading words
    r"[A-Z][A-Za-z&\./]{0,25}[ \t]*"             # final word before suffix
    r"(?:" + _ORG_SUFFIXES + r")\b",
    re.M
)

# Also match known document-specific org names by pattern
_KNOWN_ORG_RE = re.compile(
    r"\b(?:"
    r"NorthRiver\s+Benefits(?:\s+Administration)?"
    r"|Blue\s+Cedar\s+Analytics(?:\s+LLC)?"
    r"|First\s+Federal\s+Treasury\s+Services"
    r"|BrightSmile\s+Dental"
    r"|CityLab\s+Diagnostics"
    r"|West\s+End\s+Imaging"
    r"|Downtown\s+Orthopedics"
    r"|Metro\s+Rehab"
    r"|North\s+Shore\s+Vision"
    r"|Lakeview\s+(?:Family\s+Practice|Rx)"
    r"|NorthRiver\s+Urgent\s+Care"
    r"|SpecialtyHub"
    r"|MailOrder\s+Central"
    r")\b",
    re.I
)

def _find_orgs(text:str) -> List[Tuple[int,int,str]]:
    results = []
    seen = set()
    for m in _ORG_RE.finditer(text):
        v = m.group().strip()
        if len(v) >= 4 and v not in seen:
            seen.add(v)
            results.append((m.start(), m.start()+len(v), v))
    for m in _KNOWN_ORG_RE.finditer(text):
        v = m.group().strip()
        if len(v) >= 4 and v not in seen:
            seen.add(v)
            results.append((m.start(), m.start()+len(v), v))
    return results

# Person name: optional title + 2-3 consecutive TitleCase words (no newlines)
_PERSON_RE = re.compile(
    r"(?:(?:" + "|".join(re.escape(t) for t in _PERSON_TITLES) + r")\.?\s+)?"
    r"[A-Z][a-z]{1,20}(?:\s+[A-Z][a-z]{1,20}){1,2}\b"
)

# Strict person-name: optional title (Dr/Mr/Mrs/Ms) + 2-3 pure-alpha TitleCase words, no newline
_STRICT_NAME_RE = re.compile(
    r'(?:(?:Dr|Mr|Mrs|Ms|Prof)\.?\s+)?'
    r'(?<![A-Za-z])([A-Z][a-z]{1,20})[ \t]+([A-Z][a-z]{1,20})(?:[ \t]+([A-Z][a-z]{1,20}))?(?![A-Za-z])'
)

def _find_persons(text:str) -> List[Tuple[int,int,str]]:
    results = []
    for m in _STRICT_NAME_RE.finditer(text):
        parts = [g for g in [m.group(1), m.group(2), m.group(3)] if g]
        # Every word must be PURELY alphabetic
        if not all(re.match(r'^[A-Za-z]+$', p) for p in parts):
            continue
        # If the optional 3rd word is a stop word, fall back to 2-word name
        if len(parts) == 3 and parts[2].lower() in _NAME_STOP:
            parts = parts[:2]
        # NONE of the remaining words can be a stop/structural word
        if any(p.lower() in _NAME_STOP for p in parts):
            continue
        if len(parts) < 2:
            continue
        v = " ".join(parts)
        results.append((m.start(), m.start() + len(v), v))
    return results

#  Orchestrator
# ══════════════════════════════════════════════════════════════════════
@dataclasses.dataclass
class PIIMatch:
    label:str; original:str; masked:str; start:int; end:int


def _overlaps(s,e, occupied):
    return any(s < oe and e > os for os,oe in occupied)


def detect_pii(text:str, faker:RealisticFaker) -> List[PIIMatch]:
    """
    Priority: Regex > ORG > Person Names > Address
    Strict non-overlap: lower-priority layers fill gaps only.
    """
    occupied: List[Tuple[int,int]] = []
    accepted = []

    def add(s:int, e:int, orig:str, label:str):
        o = orig.strip()
        if not o or len(o) < 2: return
        lead = len(orig) - len(orig.lstrip())
        ss, ee = s + lead, s + lead + len(o)
        if not _overlaps(ss, ee, occupied):
            accepted.append((ss, ee, o, label))
            occupied.append((ss, ee))

    # Layer 1 — Regex (highest priority)
    spans = [(m.start(), m.end(), m.group(), lbl)
             for lbl,rx in _COMPILED for m in rx.finditer(text)]
    spans.sort(key=lambda x:(x[0],-(x[1]-x[0])))
    for s,e,v,l in spans:
        add(s,e,v,l)

    # Layer 2 — ORG
    for s,e,v in sorted(_find_orgs(text), key=lambda x:(x[0],-(x[1]-x[0]))):
        add(s,e,v,"ORG")

    # Layer 3 — Address (before names so street words don't get caught as names)
    for s,e,v in sorted(_find_addresses(text), key=lambda x:(x[0],-(x[1]-x[0]))):
        add(s,e,v,"ADDRESS")

    # Layer 4 — Person names (fills remaining gaps)
    for s,e,v in sorted(_find_persons(text), key=lambda x:(x[0],-(x[1]-x[0]))):
        add(s,e,v,"NAME")

    accepted.sort(key=lambda x:x[0])
    matches = []
    for s,e,orig,lbl in accepted:
        masked = faker.mask(orig, lbl)
        matches.append(PIIMatch(lbl, orig, masked, s, e))
        log.debug("  [%-16s] %-35r → %r", lbl, orig, masked)
    return matches


# ══════════════════════════════════════════════════════════════════════
#  Word Boxes + Visual Redaction  (token-match from v3)
# ══════════════════════════════════════════════════════════════════════
@dataclasses.dataclass
class WBox:
    text:str; x0:float; top:float; x1:float; bottom:float

def _get_words(page) -> List[WBox]:
    return [WBox(w["text"],w["x0"],w["top"],w["x1"],w["bottom"])
            for w in page.extract_words(keep_blank_chars=False, use_text_flow=True)]

def _norm(s:str) -> str:
    return re.sub(r"[\s\-./,()+]","",s).lower()

def find_boxes(value:str, words:List[WBox]) -> List[WBox]:
    if not value.strip(): return []
    vn = _norm(value)
    if not vn: return []

    # Pass 1: single word exact / normalized
    for w in words:
        if w.text.strip() == value.strip() or _norm(w.text) == vn:
            return [w]

    # Pass 2: sliding window of 1-8 words
    for win in range(1, min(9, len(words)+1)):
        for i in range(len(words)-win+1):
            grp = words[i:i+win]
            combined = " ".join(w.text for w in grp)
            cn = _norm(combined)
            if not cn or not vn or len(cn) < 2:
                continue
            # Require substantial overlap — no short words matching inside long values
            quality_fwd = vn in cn                                    # value in combined
            quality_rev = cn in vn and len(cn) >= max(4, len(vn)//2)  # combined in value (must be ≥50%)
            exact_raw   = value.strip() in combined
            if quality_fwd or quality_rev or exact_raw:
                tops = [w.top for w in grp]
                if max(tops)-min(tops) < 22:
                    return grp

    # Pass 3: token matching
    vtoks = [t for t in re.split(r"[\s]+", value.strip()) if len(t)>2]
    if len(vtoks) >= 2:
        found, pool = [], {i:w for i,w in enumerate(words)}
        for vt in vtoks:
            vtn = _norm(vt)
            for i,w in list(pool.items()):
                if _norm(w.text) == vtn or vtn in _norm(w.text):
                    found.append(w); del pool[i]; break
        if len(found) >= max(1, len(vtoks)//2):
            tops = [b.top for b in found]
            if not found or max(tops)-min(tops) < 30:
                return found

    # Pass 4: significant substring (require wc to cover ≥30% of vc length)
    vc = value.strip().replace(" ","")
    for w in words:
        wc = w.text.replace(" ","")
        if len(wc) >= 4 and (vc in wc or (wc in vc and len(wc) >= max(4, len(vc)//3))):
            return [w]

    return []

def _font(size:int) -> ImageFont.ImageFont:
    for p in ["/usr/share/fonts/truetype/dejavu/DejaVuSansMono.ttf",
              "/usr/share/fonts/truetype/liberation/LiberationMono-Regular.ttf",
              "/usr/share/fonts/truetype/freefont/FreeMono.ttf"]:
        try: return ImageFont.truetype(p, size)
        except: pass
    return ImageFont.load_default()

def redact(img:Image.Image, boxes:List[WBox], text:str, scale:float) -> Image.Image:
    if not boxes: return img
    draw = ImageDraw.Draw(img)
    PAD = 3
    x0 = min(b.x0 for b in boxes); y0 = min(b.top for b in boxes)
    x1 = max(b.x1 for b in boxes); y1 = max(b.bottom for b in boxes)
    px0=int(x0*scale)-PAD; py0=int(y0*scale)-PAD
    px1=int(x1*scale)+PAD; py1=int(y1*scale)+PAD
    draw.rectangle([px0,py0,px1,py1], fill=(245,245,245))
    if text.strip():
        bh = py1-py0; bw = px1-px0
        sz = max(7, int(bh*0.74))
        for s in range(sz, 6, -1):
            f = _font(s)
            try: tw = draw.textbbox((0,0), text.strip(), font=f)[2]
            except: tw = len(text)*s*0.6
            if tw <= bw+4: sz=s; break
        draw.text((px0+1, py0+1), text.strip(), fill=(25,25,25), font=_font(sz))
    return img


# ══════════════════════════════════════════════════════════════════════
#  Main PDF Pipeline
# ══════════════════════════════════════════════════════════════════════
def process_pdf(input_path:str, output_path:str,
                scale:float=2.0, seed:int=42, report:bool=True):

    input_path  = Path(input_path)
    output_path = Path(output_path)
    faker = RealisticFaker(seed=seed)
    log.info("Opening: %s  (scale=%.1f)", input_path, scale)

    plumber_pdf = pdfplumber.open(str(input_path))
    pdfium_doc  = pdfium.PdfDocument(str(input_path))
    page_images = []
    summary     = {}
    total_det   = 0
    total_drawn = 0

    for idx, pl_page in enumerate(plumber_pdf.pages):
        log.info("  Page %d/%d", idx+1, len(plumber_pdf.pages))
        img = pdfium_doc[idx].render(scale=scale, rotation=0).to_pil().convert("RGB")

        raw = pl_page.extract_text(layout=False) or ""
        # Rejoin line-broken emails
        text = re.sub(r'([A-Za-z0-9._%+\-]+@[A-Za-z0-9.\-]+)\.\n([A-Za-z]{2,})',
                      lambda m: m.group(1)+"."+m.group(2), raw)

        if not text.strip():
            page_images.append(img); continue

        matches = detect_pii(text, faker)
        summary[idx+1] = matches
        total_det += len(matches)

        if not matches:
            page_images.append(img); continue

        words = _get_words(pl_page)
        drawn = 0

        for m in matches:
            boxes = find_boxes(m.original, words)
            if not boxes:
                # fallback: find word containing key digits/letters
                sig = re.findall(r'[A-Za-z0-9]{4,}', m.original)
                for run in sig:
                    for w in words:
                        if run.lower() in w.text.lower() and len(w.text)>=3:
                            boxes=[w]; break
                    if boxes: break
            if boxes:
                img = redact(img, boxes, m.masked, scale)
                drawn += 1
                log.debug("    [%-16s] %-32r → %-32r  (%d boxes)",
                          m.label, m.original, m.masked, len(boxes))
            else:
                log.warning("    NO BOX: [%s] %r", m.label, m.original)

        total_drawn += drawn
        log.info("    detected=%d  drawn=%d/%d", len(matches), drawn, len(matches))
        page_images.append(img)

    plumber_pdf.close()
    pdfium_doc.close()

    log.info("Saving %s", output_path)
    rgb = [i.convert("RGB") for i in page_images]
    rgb[0].save(str(output_path), save_all=True, append_images=rgb[1:], resolution=150)
    log.info("DONE — detected=%d  drawn=%d  pages=%d",
             total_det, total_drawn, len(page_images))

    if report: _print_report(summary, total_drawn, total_det)
    return summary


def _print_report(summary, drawn, total):
    print("\n"+"═"*72)
    print("  PDF PII MASKING REPORT v4")
    print("  Targets: NAME · ORG · ADDRESS · PHONE · EMAIL · SSN · ACCOUNT · MORE")
    print("═"*72)
    counts: Dict[str,int] = {}
    grand = 0
    for pg, matches in sorted(summary.items()):
        if matches:
            print(f"\n  Page {pg}:")
            for m in matches:
                print(f"    [{m.label:<16}] {m.original!r:<38} → {m.masked!r}")
                counts[m.label] = counts.get(m.label,0)+1
                grand += 1
    print("\n"+"─"*72)
    print(f"  Total detected:             {grand}")
    print(f"  Visually drawn on image:    {drawn}")
    print(f"\n  Breakdown by type:")
    for lbl,cnt in sorted(counts.items(),key=lambda x:-x[1]):
        bar = "█" * min(cnt, 40)
        print(f"    {lbl:<22} {cnt:>3}  {bar}")
    print("═"*72+"\n")


# ══════════════════════════════════════════════════════════════════════
#  CLI
# ══════════════════════════════════════════════════════════════════════
def main():
    ap = argparse.ArgumentParser(description="PDF PII Masker v4")
    ap.add_argument("input")
    ap.add_argument("output")
    ap.add_argument("--scale",    type=float, default=2.0)
    ap.add_argument("--seed",     type=int,   default=42)
    ap.add_argument("--no-report",action="store_true")
    ap.add_argument("--debug",    action="store_true")
    args = ap.parse_args()
    if args.debug: logging.getLogger().setLevel(logging.DEBUG)
    if not Path(args.input).exists():
        log.error("Not found: %s", args.input); sys.exit(1)
    process_pdf(args.input, args.output, args.scale, args.seed, not args.no_report)

if __name__ == "__main__":
    main()
