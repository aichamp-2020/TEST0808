PDF PII Masker v5
=================
Fixes over v4:
  1. FONT MATCHING — reads per-char font size/bold from pdfplumber, draws masked
     text in LiberationSans (metric-compatible Helvetica) at the exact same pt size
     converted to pixels, so replaced text is visually indistinguishable in weight
     and size from the original.

  2. PRECISION DETECTION — rebuilt ORG layer uses strict suffix-only anchoring with
     a comprehensive label-context stoplist. Added LABEL:VALUE context detector so
     "Submitted By: John Smith" / "Policyholder: Aarav Mehta" patterns are caught.
     ORG spans no longer capture their surrounding label text.

Targets: NAME · ORG · ADDRESS · PHONE · EMAIL · SSN · ACCOUNT · DOB · EIN · NPI

# ══════════════════════════════════════════════════════════════════════
#  Font resolution — LiberationSans = metric Helvetica replacement
# ══════════════════════════════════════════════════════════════════════
_FONT_PATHS = {
    ("regular", False): "/usr/share/fonts/truetype/liberation/LiberationSans-Regular.ttf",
    ("bold",    False): "/usr/share/fonts/truetype/liberation/LiberationSans-Bold.ttf",
    ("regular", True):  "/usr/share/fonts/truetype/liberation/LiberationSans-Regular.ttf",
    ("bold",    True):  "/usr/share/fonts/truetype/liberation/LiberationSans-Bold.ttf",
}
_FALLBACK_FONT = "/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf"
_FONT_CACHE: Dict[Tuple, ImageFont.ImageFont] = {}

def get_font(size_px: int, bold: bool = False) -> ImageFont.ImageFont:
    key = (size_px, bold)
    if key in _FONT_CACHE:
        return _FONT_CACHE[key]
    path = _FONT_PATHS.get(("bold" if bold else "regular", False), _FALLBACK_FONT)
    try:
        f = ImageFont.truetype(path, max(6, size_px))
    except Exception:
        f = ImageFont.load_default()
    _FONT_CACHE[key] = f
    return f


# ══════════════════════════════════════════════════════════════════════
#  Per-word font metrics extracted from pdfplumber char stream
# ══════════════════════════════════════════════════════════════════════
@dataclasses.dataclass
class WBox:
    text: str
    x0: float; top: float; x1: float; bottom: float
    font_size: float = 9.0   # pt size from PDF char stream
    bold: bool = False

def _get_words(page) -> List[WBox]:
    """Extract words with bounding boxes AND font metrics from char stream."""
    # Build char-level font map keyed by (x, y) bucket
    char_map: Dict[Tuple[int,int], Tuple[float,bool]] = {}
    for c in page.chars:
        xk = int(c["x0"])
        yk = int(c["top"])
        fname = c.get("fontname","")
        size  = float(c.get("size", 9.0))
        bold  = "Bold" in fname or "bold" in fname
        char_map[(xk, yk)] = (size, bold)

    words = []
    for w in page.extract_words(keep_blank_chars=False, use_text_flow=True):
        # Look up font by word's top-left char position
        xk = int(w["x0"])
        yk = int(w["top"])
        size, bold = 9.0, False
        for dx in range(0, 5):
            for dy in range(0, 5):
                entry = char_map.get((xk+dx, yk+dy))
                if entry:
                    size, bold = entry
                    break
            if size != 9.0:
                break
        words.append(WBox(
            text=w["text"], x0=w["x0"], top=w["top"],
            x1=w["x1"], bottom=w["bottom"],
            font_size=size, bold=bold
        ))
    return words

class RealisticFaker:
    def __init__(self, seed=42):
        self._seed = seed
        self._cache: Dict[str,str] = {}

    def _rng(self, key:str) -> random.Random:
        h = int(hashlib.md5((str(self._seed)+key).encode()).hexdigest(),16)
        return random.Random(h & 0xFFFFFFFF)

    def _fit(self, word:str, n:int) -> str:
        w = word.strip() or "X"
        if len(w) >= n: return w[:n].capitalize()
        pad = "abcdefghijklmnopqrstuvwxyz"
        while len(w) < n: w += pad[len(w)%len(pad)]
        return w.capitalize()

    def _fitlc(self, s:str, n:int) -> str:
        s = s.strip() or "x"
        if len(s) >= n: return s[:n]
        while len(s) < n: s += s[-1]
        return s

    def fake_name(self, orig:str) -> str:
        rng = self._rng(orig)
        return " ".join(self._fit(rng.choice(_FIRST+_LAST), len(p))
                        for p in orig.split())

    def fake_org(self, orig:str) -> str:
        rng = self._rng(orig)
        pool = _ORG_PFX + _ORG_MID + _ORG_TYP + _CITIES
        parts = orig.split()
        return " ".join(
            p if p.upper() in ("LLC","INC","LTD","CORP","LP","LLP","PC","PLLC")
            else self._fit(rng.choice(pool), len(p))
            for p in parts
        )

    def fake_ssn(self, orig:str) -> str:
        rng = self._rng(orig)
        sep = "." if "." in orig else (" " if " " in orig and "-" not in orig else "-")
        d = f"9{rng.randint(10,99):02d}{sep}{rng.randint(10,99):02d}{sep}{rng.randint(1000,9999):04d}"
        return _sl(d, orig)

    def fake_email(self, orig:str) -> str:
        rng = self._rng(orig)
        clean = re.split(r'\n|\s{2,}', orig)[0].strip()
        m = re.match(r"(.+)@(.+)\.([A-Za-z]{2,})", clean)
        if not m: return self._scramble(orig)
        local = self._fitlc(
            f"{rng.choice(_FIRST).lower()}.{rng.choice(_LAST).lower()}", len(m.group(1)))
        dom = self._fitlc(rng.choice(_EDOMAINS), len(m.group(2)))
        return _sl(f"{local}@{dom}.{m.group(3)}", orig)

    def fake_phone(self, orig:str) -> str:
        rng = self._rng(orig)
        new = list(f"{rng.randint(201,989):03d}{rng.randint(200,999):03d}{rng.randint(1000,9999):04d}")
        res = list(orig); di = 0
        for i,c in enumerate(res):
            if c.isdigit() and di<len(new): res[i]=new[di]; di+=1
        return "".join(res)

    def fake_address(self, orig:str) -> str:
        rng = self._rng(orig)
        m = re.match(r"([A-Za-z ]+),\s*([A-Z]{2})\s+(\d{5}(?:-\d{4})?)", orig)
        if m:
            fc = self._fit(rng.choice(_CITIES), len(m.group(1).strip()))
            fs = rng.choice(_STATES_L)
            fz = self.fake_zip(m.group(3))
            return _sl(f"{fc}, {fs} {fz}", orig)
        toks = orig.split()
        out = []
        for i,t in enumerate(toks):
            if i==0 and re.match(r'^\d+$',t):
                n = rng.randint(10**(max(1,len(t)-1)), min(10**len(t)-1,99999))
                out.append(_sl(str(n),t))
            else:
                out.append(self._fit(rng.choice(_STREETS+_STYPES+_CITIES), len(t)))
        return " ".join(out)

    def fake_account(self, orig:str) -> str:
        rng = self._rng(orig)
        res = list(orig)
        for i,c in enumerate(res):
            if c.isdigit(): res[i]=str(rng.randint(0,9))
        return "".join(res)

    def fake_date(self, orig:str) -> str:
        rng = self._rng(orig)
        y,m,d = rng.randint(1955,2005), rng.randint(1,12), rng.randint(1,28)
        if re.match(r"\d{4}[-/]\d{2}[-/]\d{2}", orig): r=f"{y:04d}-{m:02d}-{d:02d}"
        elif re.match(r"\d{2}[/\-]\d{2}[/\-]\d{4}", orig): r=f"{m:02d}/{d:02d}/{y:04d}"
        else: r=f"{y}-{m:02d}-{d:02d}"
        return _sl(r, orig)

    def fake_zip(self, orig:str) -> str:
        rng = self._rng(orig)
        res=list(orig)
        for i,c in enumerate(res):
            if c.isdigit(): res[i]=str(rng.randint(0,9))
        return "".join(res)

    def _scramble(self, orig:str) -> str:
        rng = self._rng(orig); res=[]
        for c in orig:
            if c.isupper():   res.append(rng.choice("BCDFGHJKLMNPQRSTVWXYZ"))
            elif c.islower(): res.append(rng.choice("bcdfghjklmnpqrstvwxyz"))
            elif c.isdigit(): res.append(str(rng.randint(2,9)))
            else: res.append(c)
        return "".join(res)

    def mask(self, orig:str, label:str) -> str:
        key = f"{label}::{orig}"
        if key in self._cache: return self._cache[key]
        fn = {"SSN":self.fake_ssn,"EMAIL":self.fake_email,"PHONE":self.fake_phone,
              "CREDIT_CARD":self.fake_account,"IP_ADDRESS":lambda o:".".join(
                  str(self._rng(o).randint(1,253)) for _ in o.split(".")),
              "DATE_OF_BIRTH":self.fake_date,"ROUTING_NUMBER":lambda o:self.fake_account(o),
              "ZIP_CODE":self.fake_zip,"ADDRESS":self.fake_address,
              "ACCOUNT_NUMBER":self.fake_account,"CLAIM_ID":self.fake_account,
              "EIN":self.fake_account,"NPI":self.fake_account,
              "NAME":self.fake_name,"ORG":self.fake_org,
              }.get(label, self._scramble)
        r = _sl(fn(orig), orig)
        self._cache[key] = r
        return r


def _sl(r:str, orig:str) -> str:
    if not r: return orig
    if len(r)==len(orig): return r
    return (r+r[-1]*(len(orig)-len(r)))[:len(orig)]

    r"\b(?:[A-Z][A-Za-z&]{1,25}\s+){0,3}[A-Z][A-Za-z&]{1,25}\s+"
    r"(?:" + "|".join(sorted(_ORG_SUFFIX_WORDS, key=len, reverse=True)) + r")\b"
)


def _find_orgs(text:str) -> List[Tuple[int,int,str]]:
    results = []
    seen: Dict[Tuple,bool] = {}

    def _add(s:int, e:int, v:str):
        v = v.strip()
        if not v or len(v) < 3: return
        # Strip leading label-context words
        words_in_v = v.split()
        # Remove leading label words
        while words_in_v and words_in_v[0].lower().rstrip(".:") in _ORG_LABEL_WORDS:
            stripped = words_in_v[0]
            words_in_v = words_in_v[1:]
            s += len(stripped) + 1
        if len(words_in_v) < 1: return
        v = " ".join(words_in_v)
        if not v: return
        # Check it's not a stop phrase
        low = v.lower()
        for sp in _ORG_STOP_PHRASES:
            if sp in low: return
        key = (s, s+len(v))
        if key not in seen:
            seen[key] = True
            results.append((s, s+len(v), v))

    for m in _KNOWN_ORGS.finditer(text):
        _add(m.start(), m.end(), m.group())
    for m in _ORG_GENERIC.finditer(text):
        _add(m.start(), m.end(), m.group())

    return results

_STRICT_NAME_RE = re.compile(
    r'(?:(?:Dr|Mr|Mrs|Ms|Prof)\.?\s+)?'
    r'(?<![A-Za-z])([A-Z][a-z]{1,20})\s+([A-Z][a-z]{1,20})(?:\s+([A-Z][a-z]{1,20}))?(?![A-Za-z])'
)


def _find_persons(text:str) -> List[Tuple[int,int,str]]:
    results = []
    for m in _STRICT_NAME_RE.finditer(text):
        parts = [g for g in [m.group(1), m.group(2), m.group(3)] if g]
        if not all(re.match(r'^[A-Za-z]+$', p) for p in parts):
            continue
        # Drop stop-word 3rd token, try 2-word fallback
        if len(parts)==3 and parts[2].lower() in _NAME_STOP:
            parts = parts[:2]
        if any(p.lower() in _NAME_STOP for p in parts):
            continue
        if len(parts) < 2:
            continue
        v = " ".join(parts)
        results.append((m.start(), m.start()+len(v), v))
    return results


# ══════════════════════════════════════════════════════════════════════
#  Detection Layer 4 — Label:Value context  (NEW in v5)
#  Catches "Submitted By: John Smith", "Policyholder: Aarav Mehta", etc.
# ══════════════════════════════════════════════════════════════════════
_NAME_LABELS = re.compile(
    r'\b(?:submitted\s+by|policyholder(?:\s+name)?|full\s+name|patient(?:\s+name)?'
    r'|insured(?:\s+name)?|member(?:\s+name)?|employee(?:\s+name)?|applicant'
    r'|primary\s+subscriber|subscriber(?:\s+name)?|dependent\s+\d+'
    r'|claimant|beneficiary|authorized\s+by|prepared\s+by|signed\s+by'
    r'|primary\s+care|pcp|rendering\s+provider|attending\s+physician'
    r'|patient|guardian|contact(?:\s+person)?|name)\s*[:\-]\s*',
    re.I
)

# Name-value: 1-3 TitleCase words after the label
_NAME_VAL_RE = re.compile(r'([A-Z][a-z]{1,20}(?:\s+[A-Z][a-z]{1,20}){0,2})')


def _find_label_names(text:str) -> List[Tuple[int,int,str]]:
    """Find names that appear as values after name-field labels."""
    results = []
    for m in _NAME_LABELS.finditer(text):
        # Look at what follows the label+colon
        rest = text[m.end():m.end()+60]
        vm = _NAME_VAL_RE.match(rest)
        if vm:
            v = vm.group(1).strip()
            words = v.split()
            # Must be 2+ words, none in stop list
            if len(words) >= 2 and not any(w.lower() in _NAME_STOP for w in words):
                s = m.end()
                results.append((s, s+len(v), v))
    return results

#  Detection Layer 5 — Address

_ADDR_SUFX = {"st","street","ave","avenue","blvd","boulevard","rd","road",
              "ln","lane","dr","drive","ct","court","pl","place","way",
              "cir","circle","hwy","highway","pkwy","parkway","ter","terrace"}
_US_ST = {"AL","AK","AZ","AR","CA","CO","CT","DE","FL","GA","HI","ID","IL","IN",
          "IA","KS","KY","LA","ME","MD","MA","MI","MN","MS","MO","MT","NE","NV",
          "NH","NJ","NM","NY","NC","ND","OH","OK","OR","PA","RI","SC","SD","TN",
          "TX","UT","VT","VA","WA","WV","WI","WY","DC"}

_STREET_RE = re.compile(
    r'\b\d{2,5}\s+(?:[A-Za-z]+\s+){0,4}'
    r'(?:'+"|".join(_ADDR_SUFX)+r')(?:\.|\b)'
    r'(?:,?\s*[A-Za-z\s]{2,25})?(?:,?\s*(?:'+"|".join(_US_ST)+r'))?'
    r'(?:\s+\d{5}(?:-\d{4})?)?', re.I
)
_CITYSTATE_RE = re.compile(
    r'\b([A-Z][a-z]{2,20}(?:\s+[A-Z][a-z]{2,20})?),\s*('
    +"|".join(_US_ST)+r')(?:\s+(\d{5}(?:-\d{4})?))?'
)
_POBOX_RE = re.compile(r'\bP\.?\s*O\.?\s*Box\s+\d{3,6}\b', re.I)


def _find_addresses(text:str) -> List[Tuple[int,int,str]]:
    results = []
    for m in _STREET_RE.finditer(text):
        results.append((m.start(), m.start()+len(m.group().strip()), m.group().strip()))
    for m in _CITYSTATE_RE.finditer(text):
        results.append((m.start(), m.end(), m.group().strip()))
    for m in _POBOX_RE.finditer(text):
        results.append((m.start(), m.end(), m.group().strip()))
    return results

#  Orchestrator
# ══════════════════════════════════════════════════════════════════════
@dataclasses.dataclass
class PIIMatch:
    label:str; original:str; masked:str; start:int; end:int


def _overlaps(s,e,occ): return any(s<oe and e>os for os,oe in occ)


def detect_pii(text:str, faker:RealisticFaker) -> List[PIIMatch]:
    occ: List[Tuple[int,int]] = []
    acc = []

    def add(s,e,orig,lbl):
        o = orig.strip()
        if not o or len(o)<2: return
        lead = len(orig)-len(orig.lstrip())
        ss,ee = s+lead, s+lead+len(o)
        if not _overlaps(ss,ee,occ):
            acc.append((ss,ee,o,lbl)); occ.append((ss,ee))

    # 1. Regex
    for lbl,rx in _COMPILED:
        for m in sorted(rx.finditer(text), key=lambda x:(x.start(),-(x.end()-x.start()))):
            add(m.start(),m.end(),m.group(),lbl)

    # 2. ORG
    for s,e,v in sorted(_find_orgs(text),key=lambda x:(x[0],-(x[1]-x[0]))):
        add(s,e,v,"ORG")

    # 3. Address (before names so street words don't become name matches)
    for s,e,v in sorted(_find_addresses(text),key=lambda x:(x[0],-(x[1]-x[0]))):
        add(s,e,v,"ADDRESS")

    # 4. Label:Value names (high-confidence — explicit field context)
    for s,e,v in sorted(_find_label_names(text),key=lambda x:(x[0],-(x[1]-x[0]))):
        add(s,e,v,"NAME")

    # 5. Person names (fills remaining gaps)
    for s,e,v in sorted(_find_persons(text),key=lambda x:(x[0],-(x[1]-x[0]))):
        add(s,e,v,"NAME")

    acc.sort(key=lambda x:x[0])
    return [PIIMatch(lbl,o,faker.mask(o,lbl),s,e) for s,e,o,lbl in acc]


# ══════════════════════════════════════════════════════════════════════
#  Box finder — token-match (from v3/v4)
# ══════════════════════════════════════════════════════════════════════
def _norm(s:str)->str: return re.sub(r"[\s\-./,()+]","",s).lower()

def find_boxes(value:str, words:List[WBox]) -> List[WBox]:
    if not value.strip(): return []
    vn = _norm(value)
    if not vn: return []

    # Pass 1: single word exact/normalized
    for w in words:
        if w.text.strip()==value.strip() or _norm(w.text)==vn:
            return [w]

    # Pass 2: sliding window
    for win in range(1, min(9,len(words)+1)):
        for i in range(len(words)-win+1):
            grp = words[i:i+win]
            combined = " ".join(w.text for w in grp)
            cn = _norm(combined)
            if not cn or len(cn)<2: continue
            if vn in cn or (cn in vn and len(cn)>=max(4,len(vn)//2)) or value.strip() in combined:
                tops=[w.top for w in grp]
                if max(tops)-min(tops)<22: return grp

    # Pass 3: token matching
    vtoks=[t for t in re.split(r'[\s]+',value.strip()) if len(t)>2]
    if len(vtoks)>=2:
        found, pool = [], {i:w for i,w in enumerate(words)}
        for vt in vtoks:
            vtn=_norm(vt)
            for i,w in list(pool.items()):
                if _norm(w.text)==vtn or vtn in _norm(w.text):
                    found.append(w); del pool[i]; break
        if len(found)>=max(1,len(vtoks)//2):
            tops=[b.top for b in found]
            if not found or max(tops)-min(tops)<30: return found

    # Pass 4: significant substring
    vc=value.strip().replace(" ","")
    for w in words:
        wc=w.text.replace(" ","")
        if len(wc)>=4 and (vc in wc or (wc in vc and len(wc)>=max(4,len(vc)//3))):
            return [w]
    return []

#  Visual Redaction — font-matched rendering  (v5 core fix)
# ══════════════════════════════════════════════════════════════════════
def redact_and_replace(img:Image.Image, boxes:List[WBox], masked_text:str,
                       scale:float, bg_color=(248,248,248)) -> Image.Image:
    """
    Whiteout the word boxes and draw masked_text in LiberationSans
    at a font size derived from the original PDF char metrics.
    This makes replaced text visually match the surrounding text.
    """
    if not boxes: return img
    draw = ImageDraw.Draw(img)
    PAD = 2

    x0 = min(b.x0 for b in boxes);   y0 = min(b.top    for b in boxes)
    x1 = max(b.x1 for b in boxes);   y1 = max(b.bottom for b in boxes)
    bold = any(b.bold for b in boxes)

    # Convert PDF pt size → screen pixels at render scale
    # PDF: 1 pt = 1/72 inch. At 72 DPI, 1pt = 1px. At scale×72 DPI, 1pt = scale px.
    pdf_pt_size = max(b.font_size for b in boxes) if boxes else 9.0
    font_px = max(6, int(pdf_pt_size * scale))

    px0=int(x0*scale)-PAD; py0=int(y0*scale)-PAD
    px1=int(x1*scale)+PAD; py1=int(y1*scale)+PAD
    box_w = px1-px0; box_h = py1-py0

    # Whiteout
    draw.rectangle([px0, py0, px1, py1], fill=bg_color)

    if not masked_text.strip(): return img

    text_to_draw = masked_text.strip()

    # Try font at derived size, shrink if text overflows box width
    for attempt in range(font_px, 5, -1):
        f = get_font(attempt, bold)
        try:
            bbox = draw.textbbox((0,0), text_to_draw, font=f)
            tw = bbox[2]-bbox[0]
        except Exception:
            tw = len(text_to_draw) * attempt * 0.55
        if tw <= box_w + 4:
            font_px = attempt
            break

    font = get_font(font_px, bold)

    # Vertical center in box
    try:
        tbbox = draw.textbbox((0,0), text_to_draw, font=font)
        th = tbbox[3]-tbbox[1]
    except Exception:
        th = font_px
    ty = py0 + max(0, (box_h - th)//2)

    draw.text((px0+1, ty), text_to_draw, fill=(15,15,15), font=font)
    return img


# ══════════════════════════════════════════════════════════════════════
#  Main Pipeline
# ══════════════════════════════════════════════════════════════════════
def process_pdf(input_path:str, output_path:str,
                scale:float=2.0, seed:int=42, report:bool=True):
    input_path  = Path(input_path)
    output_path = Path(output_path)
    faker = RealisticFaker(seed=seed)
    log.info("Opening: %s  scale=%.1f", input_path, scale)

    plumber_pdf = pdfplumber.open(str(input_path))
    pdfium_doc  = pdfium.PdfDocument(str(input_path))
    pages_out, summary = [], {}
    total_det = total_drawn = 0

    for idx, pl_page in enumerate(plumber_pdf.pages):
        log.info("  Page %d/%d", idx+1, len(plumber_pdf.pages))
        img = pdfium_doc[idx].render(scale=scale, rotation=0).to_pil().convert("RGB")

        raw = pl_page.extract_text(layout=False) or ""
        text = re.sub(r'([A-Za-z0-9._%+\-]+@[A-Za-z0-9.\-]+)\.\n([A-Za-z]{2,})',
                      lambda m: m.group(1)+"."+m.group(2), raw)

        if not text.strip():
            pages_out.append(img); continue

        matches = detect_pii(text, faker)
        summary[idx+1] = matches
        total_det += len(matches)

        words = _get_words(pl_page)
        drawn = 0

        for m in matches:
            boxes = find_boxes(m.original, words)
            if not boxes:
                for run in re.findall(r'[A-Za-z0-9]{4,}', m.original):
                    for w in words:
                        if run.lower() in w.text.lower() and len(w.text)>=3:
                            boxes=[w]; break
                    if boxes: break
            if boxes:
                img = redact_and_replace(img, boxes, m.masked, scale)
                drawn += 1
                log.debug("    [%-16s] %r → %r", m.label, m.original, m.masked)
            else:
                log.warning("    NO BOX [%s]: %r", m.label, m.original)

        total_drawn += drawn
        log.info("    detected=%d  drawn=%d/%d", len(matches), drawn, len(matches))
        pages_out.append(img)

    plumber_pdf.close(); pdfium_doc.close()

    log.info("Saving → %s", output_path)
    rgb = [i.convert("RGB") for i in pages_out]
    rgb[0].save(str(output_path), save_all=True, append_images=rgb[1:], resolution=150)
    log.info("DONE — detected=%d  drawn=%d  pages=%d",
             total_det, total_drawn, len(pages_out))

    if report: _print_report(summary, total_drawn, total_det)
    return summary


def _print_report(summary, drawn, total):
    print("\n"+"═"*72)
    print("  PII MASKING REPORT v5 — Font-Matched · Precision Detection")
    print("═"*72)
    counts:Dict[str,int]={}
    grand=0
    for pg,matches in sorted(summary.items()):
        if matches:
            print(f"\n  Page {pg}:")
            for m in matches:
                lo=len(m.original.strip()); lm=len(m.masked.strip())
                ok="✓" if lo==lm else f"✗{lo}→{lm}"
                print(f"    [{m.label:<16}] {m.original!r:<36} → {m.masked!r}  [{ok}]")
                counts[m.label]=counts.get(m.label,0)+1; grand+=1
    print("\n"+"─"*72)
    print(f"  Total: {grand}   Drawn: {drawn}")
    for lbl,cnt in sorted(counts.items(),key=lambda x:-x[1]):
        print(f"    {lbl:<22} {cnt:>3}  {'█'*min(cnt,35)}")
    print("═"*72+"\n")


def main():
    ap=argparse.ArgumentParser(description="PDF PII Masker v5")
    ap.add_argument("input"); ap.add_argument("output")
    ap.add_argument("--scale",type=float,default=2.0)
    ap.add_argument("--seed",type=int,default=42)
    ap.add_argument("--no-report",action="store_true")
    ap.add_argument("--debug",action="store_true")
    args=ap.parse_args()
    if args.debug: logging.getLogger().setLevel(logging.DEBUG)
    if not Path(args.input).exists():
        log.error("Not found: %s",args.input); sys.exit(1)
    process_pdf(args.input,args.output,args.scale,args.seed,not args.no_report)

if __name__=="__main__":
    main()
