"""
PDF PII Masker v7 — Hybrid Text + Image Masking (Windows/Mac/Linux)
====================================================================
Three-mode pipeline handles every PDF type:

  MODE 1 — PURE TEXT PAGE  (chars >= 20, no embedded images)
    Detection : pdfplumber.extract_text()
    Positioning: Spatial char-stream word groups → exact pixel coords
    Rendering  : LiberationSans/Arial at matched PDF pt size + color

  MODE 2 — HYBRID PAGE  (chars >= 20 AND embedded images present)
    → Runs MODE 1 on the selectable text layer
    → ALSO runs OCR on every embedded image region
    → Masks both text layer and image regions independently

  MODE 3 — PURE IMAGE PAGE  (chars < 20)
    Detection : pytesseract OCR on full page render
    Rendering  : Same OCR bbox masking

PII TARGETS: Name · ORG · Address · Phone · Email · SSN
             Account IDs · DOB · EIN · NPI · Routing · ZIP

INSTALL:
  pip install pdfplumber pypdfium2 Pillow pytesseract pypdf

Tesseract (for image/hybrid pages):
  Windows : https://github.com/UB-Mannheim/tesseract/wiki
  macOS   : brew install tesseract
  Linux   : sudo apt install tesseract-ocr

USAGE:
  python pdf_pii_masker_v7.py input.pdf output.pdf
  python pdf_pii_masker_v7.py input.pdf output.pdf --scale 3.0
  python pdf_pii_masker_v7.py input.pdf output.pdf --debug
"""

# ─── Tesseract setup ──────────────────────────────────────────────────
def _setup_tess() -> bool:
    if not _HAVE_TESS: return False
    candidates = [
        os.path.join(os.environ.get("PROGRAMFILES","C:\\Program Files"),"Tesseract-OCR","tesseract.exe"),
        os.path.join(os.environ.get("PROGRAMFILES(X86)",""),"Tesseract-OCR","tesseract.exe"),
        os.path.expanduser("~\\AppData\\Local\\Tesseract-OCR\\tesseract.exe"),
        "/usr/local/bin/tesseract", "/opt/homebrew/bin/tesseract", "/usr/bin/tesseract",
    ]
    for p in candidates:
        if os.path.isfile(p):
            pytesseract.pytesseract.tesseract_cmd = p
            log.info("Tesseract: %s", p); return True
    t = shutil.which("tesseract")
    if t:
        pytesseract.pytesseract.tesseract_cmd = t
        log.info("Tesseract (PATH): %s", t); return True
    log.warning("Tesseract not found — image PII on embedded images will not be masked.\n"
        "  Windows: https://github.com/UB-Mannheim/tesseract/wiki\n"
        "  macOS:   brew install tesseract\n  Linux:   sudo apt install tesseract-ocr")
    return False

# ─── Font ──────────────────────────────────────────────────────────────
def _find_font(bold=False):
    wf = os.path.join(os.environ.get("WINDIR","C:\\Windows"), "Fonts")
    cands = (
        [os.path.join(wf,"arialbd.ttf" if bold else "arial.ttf"),
         os.path.join(wf,"calibrib.ttf" if bold else "calibri.ttf")]
        if platform.system()=="Windows" else
        ["/System/Library/Fonts/Helvetica.ttc",
         "/Library/Fonts/Arial Bold.ttf" if bold else "/Library/Fonts/Arial.ttf"]
        if platform.system()=="Darwin" else
        ["/usr/share/fonts/truetype/liberation/LiberationSans-Bold.ttf" if bold
         else "/usr/share/fonts/truetype/liberation/LiberationSans-Regular.ttf",
         "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf" if bold
         else "/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf"]
    )
    for p in cands:
        if os.path.isfile(p): return p
    return None

_FC = {}
def _font(px, bold=False):
    k=(px,bold)
    if k not in _FC:
        p=_find_font(bold); px=max(6,px)
        try:   _FC[k]=ImageFont.truetype(p,px) if p else ImageFont.load_default()
        except: _FC[k]=ImageFont.load_default()
    return _FC[k]

# ─── Spatial word (char-stream) ────────────────────────────────────────
@dataclasses.dataclass
class SW:
    text:str; x0:float; top:float; x1:float; bottom:float
    size:float; bold:bool; color:Tuple

def _sw(page) -> List[SW]:
    raw=page.chars
    if not raw: return []
    grps,cur=[],[]
    for c in raw:
        if not c["text"] or not c["text"].strip():
            if cur: grps.append(cur); cur=[]; continue
        if not cur: cur.append(c); continue
        if abs(c["top"]-cur[-1]["top"])<3 and c["x0"]-cur[-1]["x1"]<3: cur.append(c)
        else:
            if cur: grps.append(cur)
            cur=[c]
    if cur: grps.append(cur)
    res=[]
    for g in grps:
        fn=g[0].get("fontname",""); bold="Bold" in fn or "bold" in fn
        col=g[0].get("non_stroking_color",(0.,0.,0.))
        if not isinstance(col,(list,tuple)): col=(0.,0.,0.)
        col=tuple(float(v) for v in list(col)[:3])
        if len(col)<3: col=col+(0.,)*(3-len(col))
        res.append(SW(text="".join(c["text"] for c in g),
            x0=min(c["x0"] for c in g),top=min(c["top"] for c in g),
            x1=max(c["x1"] for c in g),bottom=max(c["bottom"] for c in g),
            size=float(g[0].get("size",9.0)),bold=bold,color=col))
    return res

def _norm(s): return re.sub(r"[\s\-./,()+@]","",s).lower()

def _find_boxes(value, words):
    if not value.strip(): return []
    vn=_norm(value)
    if not vn: return []
    for w in words:
        if w.text==value.strip() or _norm(w.text)==vn: return [w]
    for win in range(1,min(9,len(words)+1)):
        for i in range(len(words)-win+1):
            grp=words[i:i+win]
            if max(w.top for w in grp)-min(w.top for w in grp)>22: continue
            combined=" ".join(w.text for w in grp); cn=_norm(combined)
            if not cn or len(cn)<2: continue
            if vn in cn or (cn in vn and len(cn)>=max(4,len(vn)//2)) or value.strip() in combined:
                return grp
    vtoks=[t for t in re.split(r"\s+",value.strip()) if len(t)>2]
    if len(vtoks)>=2:
        found=[]; pool={i:w for i,w in enumerate(words)}
        for vt in vtoks:
            vtn=_norm(vt)
            for i,w in list(pool.items()):
                if _norm(w.text)==vtn or vtn in _norm(w.text): found.append(w); del pool[i]; break
        if len(found)>=max(1,len(vtoks)//2):
            tops=[b.top for b in found]
            if not found or max(tops)-min(tops)<30: return found
    vc=value.strip().replace(" ","")
    for w in words:
        wc=w.text.replace(" ","")
        if len(wc)>=4 and (vc in wc or (wc in vc and len(wc)>=max(4,len(vc)//3))): return [w]
    return []

# ─── Redact text layer ────────────────────────────────────────────────
def _c255(c): return tuple(min(255,max(0,int(v*255))) for v in c[:3])

def redact_text(img, boxes, masked, scale):
    if not boxes: return img
    draw=ImageDraw.Draw(img); PAD=1
    x0=min(b.x0 for b in boxes); y0=min(b.top for b in boxes)
    x1=max(b.x1 for b in boxes); y1=max(b.bottom for b in boxes)
    fpx=max(6,int(boxes[0].size*scale)); bld=boxes[0].bold; tc=_c255(boxes[0].color)
    px0=int(x0*scale)-PAD; py0=int(y0*scale)-PAD
    px1=int(x1*scale)+PAD; py1=int(y1*scale)+PAD
    bw=px1-px0; bh=py1-py0
    draw.rectangle([px0,py0,px1,py1],fill=(255,255,255))
    if not masked.strip(): return img
    for a in range(fpx,5,-1):
        f=_font(a,bld)
        try: tw=draw.textbbox((0,0),masked.strip(),font=f)[2]
        except: tw=len(masked)*a*0.55
        if tw<=bw+6: fpx=a; break
    f=_font(fpx,bld)
    try: th=draw.textbbox((0,0),masked.strip(),font=f)[3]
    except: th=fpx
    draw.text((px0+1,py0+max(0,(bh-th)//2)),masked.strip(),fill=tc,font=f)
    return img

# ─── Redact image region (OCR bbox) ──────────────────────────────────
def redact_ocr_box(img, x0, y0, x1, y1, masked, bg_sample_color=None):
    """White-out OCR bbox and draw masked text at matched size."""
    draw=ImageDraw.Draw(img)
    bw=x1-x0; bh=y1-y0
    bg = bg_sample_color or (255,255,255)
    draw.rectangle([x0-1,y0-1,x1+1,y1+1],fill=bg)
    if not masked.strip(): return img
    sz=max(6,int(bh*0.72))
    for a in range(sz,5,-1):
        f=_font(a,False)
        try: tw=draw.textbbox((0,0),masked,font=f)[2]
        except: tw=len(masked)*a*0.55
        if tw<=bw+4: sz=a; break
    f=_font(sz,False)
    try: th=draw.textbbox((0,0),masked,font=f)[3]
    except: th=sz
    ty=y0+max(0,(bh-th)//2)
    # Use dark text on light background
    text_color=(15,15,15) if sum(bg)>380 else (240,240,240)
    draw.text((x0+2,ty),masked,fill=text_color,font=f)
    return img

def _orgs(text):
    seen={}; res=[]
    def add(s,e,v):
        v=v.strip()
        if not v: return
        ws=v.split()
        while ws and ws[0].lower().rstrip(".:") in _OSTRIP: s+=len(ws[0])+1; ws=ws[1:]
        if not ws: return
        v=" ".join(ws); lv=v.lower()
        for sp in _OSTOP:
            if lv.startswith(sp) or lv==sp: return
        k=(s,s+len(v))
        if k not in seen: seen[k]=True; res.append((s,s+len(v),v))
    for m in _KNOWN.finditer(text): add(m.start(),m.end(),m.group())
    for m in _OGEN.finditer(text): add(m.start(),m.end(),m.group())
    return res

_NRE=re.compile(r"(?:(?:Dr|Mr|Mrs|Ms|Prof)\.?\s+)?(?<![A-Za-z])([A-Z][a-z]{1,20})\s+"
    r"([A-Z][a-z]{1,20})(?:\s+([A-Z][a-z]{1,20}))?(?![A-Za-z])")
_ADDR_S={"st","street","ave","avenue","blvd","boulevard","rd","road","ln","lane",
    "dr","drive","ct","court","pl","place","way","cir","circle","hwy","highway","pkwy","ter","terrace"}
_USS={"AL","AK","AZ","AR","CA","CO","CT","DE","FL","GA","HI","ID","IL","IN","IA","KS",
    "KY","LA","ME","MD","MA","MI","MN","MS","MO","MT","NE","NV","NH","NJ","NM","NY",

_STRE=re.compile(r"\b\d{2,5}\s+(?:[A-Za-z]+\s+){0,4}(?:"+"|".join(_ADDR_S)+r")(?:\.|\b)"
    r"(?:,?\s*[A-Za-z\s]{2,25})?(?:,?\s*(?:"+"|".join(_USS)+r"))?(?:\s+\d{5}(?:-\d{4})?)?",re.I)
_CSTE=re.compile(r"\b([A-Z][a-z]{2,20}(?:\s+[A-Z][a-z]{2,20})?),\s*("+"|".join(_USS)+r")(?:\s+(\d{5}(?:-\d{4})?))?")
_PBXE=re.compile(r"\bP\.?\s*O\.?\s*Box\s+\d{3,6}\b",re.I)

def _lbn(text):
    res=[]
    for m in _LBPAT.finditer(text):
        rest=text[m.end():m.end()+60]; vm=_LBVAL.match(rest)
        if vm:
            v=vm.group(1).strip(); ws=v.split()
            if len(ws)>=2 and not any(w.lower() in _NSTOP for w in ws):
                res.append((m.end(),m.end()+len(v),v))
    return res

def _pers(text):
    res=[]
    for m in _NRE.finditer(text):
        parts=[g for g in [m.group(1),m.group(2),m.group(3)] if g]
        if not all(re.match(r"^[A-Za-z]+$",p) for p in parts): continue
        if len(parts)==3 and parts[2].lower() in _NSTOP: parts=parts[:2]
        if any(p.lower() in _NSTOP for p in parts): continue
        if len(parts)<2: continue
        v=" ".join(parts); res.append((m.start(),m.start()+len(v),v))
    return res

def _addrs(text):
    res=[]
    for m in _STRE.finditer(text): res.append((m.start(),m.start()+len(m.group().strip()),m.group().strip()))
    for m in _CSTE.finditer(text): res.append((m.start(),m.end(),m.group().strip()))
    for m in _PBXE.finditer(text): res.append((m.start(),m.end(),m.group().strip()))
    return res

@dataclasses.dataclass
class PIIMatch:
    label:str; original:str; masked:str; start:int; end:int

def _ov(s,e,occ): return any(s<oe and e>os for os,oe in occ)

def detect_pii(text, faker):
    occ=[]; acc=[]
    def add(s,e,o,l):
        orig=o.strip()
        if not orig or len(orig)<2: return
        ld=len(o)-len(o.lstrip()); ss,ee=s+ld,s+ld+len(orig)
        if not _ov(ss,ee,occ): acc.append((ss,ee,orig,l)); occ.append((ss,ee))
    for l,rx in _CRX:
        for m in sorted(rx.finditer(text),key=lambda x:(x.start(),-(x.end()-x.start()))): add(m.start(),m.end(),m.group(),l)
    for s,e,v in sorted(_orgs(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"ORG")
    for s,e,v in sorted(_addrs(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"ADDRESS")
    for s,e,v in sorted(_lbn(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"NAME")
    for s,e,v in sorted(_pers(text),key=lambda x:(x[0],-(x[1]-x[0]))): add(s,e,v,"NAME")
    acc.sort(key=lambda x:x[0])
    return [PIIMatch(l,o,faker.mask(o,l),s,e) for s,e,o,l in acc]

# ─── OCR masker — for image regions ──────────────────────────────────
def _ocr_region(img, faker, ox=0, oy=0, label=""):
    """
    Run OCR on img (or a sub-region), detect PII, draw replacements.
    ox/oy are pixel offsets for sub-region crops (for embedded images on text pages).
    label is just for logging.
    """
    if not _TESS_OK:
        log.warning("Tesseract unavailable — %s skipped", label or "image region")
        return img

    import pytesseract
    draw = ImageDraw.Draw(img)

    # Get word-level OCR data
    data = pytesseract.image_to_data(
        img if (ox==0 and oy==0) else img.crop((ox,oy,img.width,img.height)),
        output_type=pytesseract.Output.DICT, lang="eng", config="--psm 6"
    )
    n = len(data["text"])
    words = [
        (data["text"][i],
         data["left"][i] + ox, data["top"][i] + oy,
         data["width"][i], data["height"][i])
        for i in range(n)
        if data["text"][i].strip() and int(data["conf"][i]) > 25
    ]
    if not words:
        return img

    full = " ".join(w[0] for w in words)
    matches = detect_pii(full, faker)

    if not matches:
        return img

    log.info("    OCR region '%s': %d words, %d PII found", label, len(words), len(matches))

    # Build word char-offset map
    cursor = 0; wpos = []
    for txt,l,t,w,h in words:
        idx = full.find(txt, cursor)
        if idx >= 0:
            wpos.append((idx, idx+len(txt), l, t, w, h))
            cursor = idx + len(txt)

    for m in matches:
        hits = [wp for wp in wpos if wp[0]<m.end and wp[1]>m.start]
        if not hits:
            continue
        x0 = min(wp[2] for wp in hits)
        y0 = min(wp[3] for wp in hits)
        x1 = max(wp[2]+wp[4] for wp in hits)
        y1 = max(wp[3]+wp[5] for wp in hits)
        bw = x1-x0; bh = y1-y0

        # Sample background color at this location for better blending
        try:
            region = img.crop((x0, y0, x1, y1))
            avg = region.resize((1,1)).getpixel((0,0))
            bg = avg[:3] if isinstance(avg, tuple) else (255,255,255)
        except:
            bg = (255,255,255)

        img = redact_ocr_box(img, x0, y0, x1, y1, m.masked, bg)
        log.debug("    OCR [%s] %r -> %r", m.label, m.original, m.masked)

    return img

# ─── Main pipeline ────────────────────────────────────────────────────
def process_pdf(inp, out, scale=2.0, seed=42, report=True):
    inp=Path(inp); out=Path(out); faker=Faker(seed)
    log.info("Opening: %s  scale=%.1f", inp, scale)

    plb = pdfplumber.open(str(inp))
    pdc = pdfium.PdfDocument(str(inp))
    pages=[]; summary={}; td=tdr=0

    for idx, pl in enumerate(plb.pages):
        log.info("  Page %d/%d", idx+1, len(plb.pages))
        img = pdc[idx].render(scale=scale, rotation=0).to_pil().convert("RGB")

        n_chars  = len(pl.chars)
        n_images = len(pl.images)

        # ── MODE 3: Pure image page ────────────────────────────────────
        if n_chars < 20:
            log.info("    MODE 3 — pure image page: full-page OCR")
            img = _ocr_region(img, faker, label=f"page{idx+1}_full")
            pages.append(img); continue

        # ── Text extraction ────────────────────────────────────────────
        raw  = pl.extract_text(layout=False) or ""
        text = re.sub(r"([A-Za-z0-9._%+\-]+@[A-Za-z0-9.\-]+)\.\n([A-Za-z]{2,})",
                      lambda m: m.group(1)+"."+m.group(2), raw)
        matches = detect_pii(text, faker)
        summary[idx+1] = matches; td += len(matches)

        # ── MODE 1/2 text-layer masking ────────────────────────────────
        mode = "MODE 2 (hybrid)" if n_images > 0 else "MODE 1 (text)"
        log.info("    %s: text_pii=%d  embedded_images=%d", mode, len(matches), n_images)

        sw = _sw(pl)
        drawn = 0
        for m in matches:
            boxes = _find_boxes(m.original, sw)
            if not boxes:
                for run in re.findall(r"[A-Za-z0-9]{4,}", m.original):
                    for w in sw:
                        if run.lower() in w.text.lower() and len(w.text)>=3: boxes=[w]; break
                    if boxes: break
            if boxes:
                img = redact_text(img, boxes, m.masked, scale); drawn+=1
                log.debug("    TEXT [%s] %r->%r", m.label, m.original, m.masked)
            else:
                log.warning("    NO BOX [%s]: %r", m.label, m.original)
        tdr += drawn

        # ── MODE 2: Also OCR every embedded image region ───────────────
        if n_images > 0 and _TESS_OK:
            ph = pl.height  # PDF coordinate height
            for img_idx, img_info in enumerate(pl.images):
                # PDF coords (origin bottom-left) → PIL pixels (origin top-left)
                # pdfplumber already gives us top/bottom in top-left coords
                x0_pdf = img_info.get("x0", 0)
                y0_pdf = img_info.get("top", 0)
                x1_pdf = img_info.get("x1", pl.width)
                y1_pdf = img_info.get("bottom", ph)

                px0 = max(0, int(x0_pdf * scale) - 5)
                py0 = max(0, int(y0_pdf * scale) - 5)
                px1 = min(img.width,  int(x1_pdf * scale) + 5)
                py1 = min(img.height, int(y1_pdf * scale) + 5)

                log.info("    OCR embedded image[%d]: bbox=(%d,%d,%d,%d) px",
                         img_idx, px0, py0, px1, py1)

                # Crop the image region, run OCR, draw masks back on full img
                region_img = img.crop((px0, py0, px1, py1)).convert("RGB")

                # Run OCR on the cropped region
                import pytesseract
                data = pytesseract.image_to_data(
                    region_img, output_type=pytesseract.Output.DICT,
                    lang="eng", config="--psm 6"
                )
                n = len(data["text"])
                words = [
                    (data["text"][i],
                     data["left"][i] + px0, data["top"][i] + py0,
                     data["width"][i], data["height"][i])
                    for i in range(n)
                    if data["text"][i].strip() and int(data["conf"][i]) > 25
                ]

                if not words:
                    log.info("    OCR: no words found in image region")
                    continue

                full_ocr = " ".join(w[0] for w in words)
                ocr_matches = detect_pii(full_ocr, faker)
                log.info("    OCR: %d words detected, %d PII items", len(words), len(ocr_matches))

                # Build word positions (relative to full page image)
                cursor=0; wpos=[]
                for txt,l,t,w,h in words:
                    fidx = full_ocr.find(txt, cursor)
                    if fidx >= 0:
                        wpos.append((fidx, fidx+len(txt), l, t, w, h))
                        cursor = fidx + len(txt)

                draw = ImageDraw.Draw(img)
                for m in ocr_matches:
                    hits=[wp for wp in wpos if wp[0]<m.end and wp[1]>m.start]
                    if not hits: continue
                    bx0=min(wp[2] for wp in hits); by0=min(wp[3] for wp in hits)
                    bx1=max(wp[2]+wp[4] for wp in hits); by1=max(wp[3]+wp[5] for wp in hits)
                    bw=bx1-bx0; bh=by1-by0

                    # Sample background for blending
                    try:
                        reg=img.crop((bx0,by0,bx1,by1))
                        avg=reg.resize((1,1)).getpixel((0,0))
                        bg=avg[:3] if isinstance(avg,tuple) else (255,255,255)
                    except: bg=(255,255,255)

                    img = redact_ocr_box(img,bx0,by0,bx1,by1,m.masked,bg)
                    tdr += 1; drawn += 1
                    log.info("    IMG-OCR [%s] %r -> %r", m.label, m.original, m.masked)

        log.info("    drawn=%d/%d", drawn, len(matches))
        pages.append(img)

    plb.close(); pdc.close()
    log.info("Saving -> %s", out)
    rgb=[i.convert("RGB") for i in pages]
    rgb[0].save(str(out), save_all=True, append_images=rgb[1:], resolution=150)
    log.info("DONE  det=%d drawn=%d pages=%d", td, tdr, len(pages))
    if report: _rpt(summary, tdr, td)
    return summary

def _rpt(s,d,t):
    print("\n"+"="*70)
    print("  PII MASKING REPORT v7  |  Text + Embedded Image OCR + Font Match")
    print("="*70)
    c={}; g=0
    for pg,ms in sorted(s.items()):
        if ms:
            print(f"\n  Page {pg}:")
            for m in ms:
                lo=len(m.original.strip()); lm=len(m.masked.strip())
                print(f"    [{m.label:<16}] {m.original!r:<36} -> {m.masked!r}  [{'OK' if lo==lm else f'LEN:{lo}>{lm}'}]")
                c[m.label]=c.get(m.label,0)+1; g+=1
    print("\n"+"-"*70); print(f"  Total: {g}   Drawn: {d}")
    for l,n in sorted(c.items(),key=lambda x:-x[1]):
        print(f"    {l:<22} {n:>3}  {'#'*min(n,35)}")
    print("="*70)
